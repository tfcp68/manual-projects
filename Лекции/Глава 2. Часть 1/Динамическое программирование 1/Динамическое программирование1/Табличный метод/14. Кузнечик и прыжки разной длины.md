```python
#python

def grig_random_step(n, steps):
    a = [1]
    for i in range(1, n+1):
        s = 0
        for j in steps:
            if j <= i:
                s += a[i - j]
        a.append(s)
    return a[n]
```

```cpp
int grig_and_any_steps(int n, int s[]) {
    if (n < 0) {
        return -1;
    }
    std::vector<int> a(n + 1, 0);
    a[0] = 1;
    for (int i = 1; i <= n; i++) {
        for (size_t j = 0; j < sizeof(s) / sizeof(*s); j++) {
            if (s[j] <= i) {
                a[i] = a[i] + a[i - s[j]];
            }
        }
    }
    return a[n];
}
```

```cpp
}
```

```
функция оптимальный_урон_при_рандомном_шаге(damage, n, steps, лягушки):
	заполняем массив W размерности n бесконечностями
	W[0] <- 0
	для i от 1 до n
		для j в steps
			если j > i, то
                следующая итерация
            если в (i - j) лягушки, то
                W[i - j] <- inf
            W[i] <- минимум из W[i], W[i - j] + damage[i - 1]	
	вернуть W[n]
```
```python
#python

def optimal_damage_back_random(damage, n, steps, frogs):
    if n in frogs:
        return inf
    W = [0] + [inf] * n
    for i in range(0, n + 1):
        if i in frogs:
            W[i] = inf
        for j in steps:
            if j > i:
                continue
            if i - j in frogs:
                W[i - j] = inf
            W[i] = min(W[i], W[i - j] + damage[i - 1])
    return W[n]
```

```cpp
//C++
//sport version
  
int opt_damage_back_random(int n, int s[], int f[], int d[]){
    int a[n + 1];
    for (int i = 0; i < n + 1; i++){
        a[i] = std::numeric_limits<int>::max();
    }
    a[0] = 0;
    for (size_t q = 0; q < sizeof( f ) / sizeof( *f ); q++){
        if (f[q] == n){
            return -1;
        }
    }
    for (int i = 0; i <= n; i++){
        for (size_t j = 0; j < sizeof( s ) / sizeof( *s ); j++){
            if (s[j] <= i){
                for (size_t q = 0; q < sizeof( f ) / sizeof( *f ); q++){
                    if (f[q] == i - s[j]){
                        a[i - s[j]] = std::numeric_limits<int>::max();
                    }
                }
                int count;
                count = a[i - s[j]] + d[i - 1];
                if (count < a[i] && count > 0){
                    a[i] = count;
                }
            }
        }
    }
    if (a[n] == std::numeric_limits<int>::max()){
        return -1;
    }
    return a[n];
}
```

```
функция оптимальный_урон_при_рандомном_шаге(damage, n, steps, лягушки):
	заполняем массив W размерности n бесконечностями
	W[0] <- 0
	для i от 0 до n
		если в i лягушки, то
			W[i] <- inf
		 для j в steps
			 count <- W[i] + d[i + j]
			 W[i + j] <- минимум из W[i + j] и count
	вернуть W[n]
```

```python
#python

from math import inf 

def optimal_damage_forward_random(damage, n, steps, frogs):
    if n in frogs:
        return inf
    W = [0] + [inf] * n
    for i in range(0, n):
        if i in frogs:
            W[i] = inf
        for j in steps:
            if j > n - i:
                continue
            count = W[i] + damage[i + j - 1]
            W[i + j] = min(count, W[i + j])
            if i + j in frogs:
                W[i + j] = inf
    return W[n]
```

```cpp
//C++
//sport version

int opt_damage_forward_random(int n, int s[ ], int f[], int d[]){
    int a[n + 1];
    for (int i = 0; i < n + 1; i++){
        a[i] = std::numeric_limits<int>::max();
    }
    a[0] = 0;
    for (size_t q = 0; q <= sizeof( f ) / sizeof( *f ); q++){
        if (f[q] == n){
            return -1;
        }
    }
    for (int i = 0; i < n; i++){
        for (size_t q = 0; q < sizeof( f ) / sizeof( *f ); q++){
            if (f[q] == i){
                a[i] = std::numeric_limits<int>::max();
            }
        }
        for (size_t j = 0; j < sizeof( s ) / sizeof( *s ); j++){
            if (s[j] <= n - i){
                int count;
                count = a[i] + d[i + s[j] - 1];
                if (count < a[i + s[j]] && count > 0){
                    a[i + s[j]] = count;
                }
            }
        }
    }
    if (a[n] == std::numeric_limits<int>::max()){
        return -1;
    }
    return a[n];
}
```
