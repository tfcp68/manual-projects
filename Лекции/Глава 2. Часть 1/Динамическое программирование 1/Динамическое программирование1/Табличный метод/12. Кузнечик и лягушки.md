Решение через ДП назад:

Столбики с лягушками пропускаются

```
Псевдокод:

//a - массив состоящий из количества способов добраться до каждого i-го столбика
//k - длина максимального прыжка
//n - столбик, до которого необходимо дойти

функция кузнечик и лягушки(n, k, лягушки)
	для i от 0 до n
		s <- 0
		r <- минимум из k и i
		для j от 1 до r
			если в a[i - j] не сидит лягушка, то
				s <- s + a[i - j]
	    a[i] <- s
	вернуть a[n]
```
```python
#python

def grig_and_frogs_back(n, k, frogs):
    if n < 0 or k < 0:
        return -1
    if 0 in frogs:
        return 0
    a = [1]
    a = a + [0] * (n + k + 1)
    for i in range(n + 1):
        r = min(k, i)
        for j in range(1, r + 1):
            a[i] += a[i - j]
        if i in frogs:
            a[i] = 0
    return a[n]
```

```cpp
        }

            min = i;
        }

                a[i] = 0;
            }
        }
    }

```

```python
#python

def grig_and_frogs_forward(n, k, frogs):
    if n < 0 or k < 0:
        return -1
    if 0 in frogs:
        return 0
    a = [1]
    a = a + [0] * (n + k + 1)
    for i in range(n + 1):
        for j in range(1, k + 1):
            if i + j not in frogs:
                a[i + j] += a[i]
    return a[n]
```

```cpp
template<typename ValueType>
ValueType grig_and_frogs_forward(ValueType n, ValueType k, ValueType frogs[]){
    if(n < 0 or k < 0){
        return -1;
    }

    ValueType a[n + k + 1];
    for (int i = 0; i <= n + k; i++){
        a[i] = 0;
    }
    a[0] = 1;
    for (int i = 0; i < n + 1; i++){
        for (size_t q = 0; q <= sizeof( frogs ) / sizeof( *frogs ); q++){
            if (frogs[q] == i){
                a[i] = 0;
            }
        }
        for (int j = 1; j < k + 1; j++){
            a[i + j] = a[i + j] + a[i];
        }
    }
    return a[n];
}
```
